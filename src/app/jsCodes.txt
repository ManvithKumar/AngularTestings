Missing number :

class Solution{
    missingNumber(array,n){
     let expSum = (n *(n+1)) / 2;
     
     let arrSum = array.reduce((acc,cur) => acc+cur,0);
     
     let missingNum = expSum - arrSum;
     
     return missingNum;
     
    }
}

//Prime Numbers 
 function isPrime(num)
    {
        if(num < 2)
        {
            return false
        }
        for(let i =2;i<=Math.sqrt(num);i++)
        {
            if(num % i === 0)
            {
                return false
            }
        }
        return true
    }

    function generatePrimeNumbers(start,end)
    {
        primeList =[];

        for(let i = start;i<= end;i++)
        {
            if(isPrime(i))
            {
                primeList.push(i)
            }
        }
        return primeList;
    }

//Armstrong Numbers

function armstrong(number) {
    var og = number;
    var result = 0;
    while (og !== 0) {
        var remainder = og % 10;
        result = result + Math.pow(remainder, 3);
        og = Math.floor(og / 10); 
    }
  if(number === result)
  {
    return "Armstrong Number"
  }
  else{
    return "Not a Armstrong Number"
  }
}

console.log(armstrong(153)); 


//Count the frequency of letters in the string
function countLetters(str) {
    const frequency = {};
    
    for (let char of str) {

        if (/^[a-zA-Z]$/.test(char))
        {
            char = char.toLowerCase();
            frequency[char] = (frequency[char] || 0) + 1;
        }
    } 
    return frequency;
}

const inputString = "Hello World!";
const letterFrequency = countLetters(inputString);

console.log(letterFrequency);

// Function to find the HCF (GCD) of two numbers
function hcf(a, b) {
    if (b === 0) {
        return a;
    } else {
        return hcf(b, a % b);
    }
}

// Function to find the LCM of two numbers
function lcm(a, b) {
    return (a * b) / hcf(a, b);
}

// Example usage
const num1 = 12;
const num2 = 15;

console.log("HCF of", num1, "and", num2, "is:", hcf(num1, num2));
console.log("LCM of", num1, "and", num2, "is:", lcm(num1, num2));
